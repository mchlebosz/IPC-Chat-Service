18getpasswd will read upto sz - 1 chars into pw, null-terminating
19the resulting string. On success, the number of characters in
20pw are returned, -1 otherwise.

22ssize_t getpasswd(char** pw, size_t sz, int mask, FILE* fp) {

24#ifdef MAXPW

30 if (!tmp) return -1;
31 memset(tmp, 0, sz); /* initialize memory to 0 */
32 *pw = (char*)tmp;
33 }

36 int c = 0;

46

48 new_kbd_mode.c_cc[VTIME] = 0;

60 fputc(mask, stdout);

63 if (31 < mask && mask < 127) {
